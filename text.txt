// Definição da classe Cliente
class Cliente {
  constructor(id, nome, endereco, telefone, email) {
    this.id = id;
    this.nome = nome;
    this.endereco = endereco;
    this.telefone = telefone;
    this.email = email;
  }
}

// Definição da classe Pedido
class Pedido {
  constructor(id, dataHora, cliente, status, total) {
    this.id = id;
    this.dataHora = dataHora;
    this.cliente = cliente;
    this.status = status;
    this.total = total;
    this.itens = []; // Array para armazenar os itens do pedido
  }

  adicionarItem(item) {
    this.itens.push(item);
  }

  calcularTotal() {
    this.total = this.itens.reduce((total, item) => total + item.total, 0);
  }
}
